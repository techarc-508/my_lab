import { useEffect, useRef, useState } from "react";
import { Languages, HelpCircle } from "lucide-react";
import { Track } from "../types/music";

interface LyricsPanelProps {
  currentTrack: Track | null;
  currentTime: number; // in seconds
  onSeek: (time: number) => void;
  theme: "dark" | "light";
}

export default function LyricsPanel({
  currentTrack,
  currentTime,
  onSeek,
  theme,
}: LyricsPanelProps) {
  const [lyricsMode, setLyricsMode] = useState<"both" | "japanese" | "english">("both");
  const containerRef = useRef<HTMLDivElement | null>(null);
  const activeLineRef = useRef<HTMLDivElement | null>(null);

  // Find the active lyric index based on current playback time
  const lyrics = currentTrack?.lyrics || [];
  let activeIndex = -1;

  for (let i = 0; i < lyrics.length; i++) {
    if (currentTime >= lyrics[i].time) {
      activeIndex = i;
    } else {
      break;
    }
  }

  // Smoothly scroll active lyric into center of view
  useEffect(() => {
    if (activeLineRef.current && containerRef.current) {
      const activeLine = activeLineRef.current;
      const container = containerRef.current;

      const activeOffsetTop = activeLine.offsetTop;
      const containerHeight = container.clientHeight;
      const activeHeight = activeLine.clientHeight;

      container.scrollTo({
        top: activeOffsetTop - containerHeight / 2 + activeHeight / 2,
        behavior: "smooth",
      });
    }
  }, [activeIndex]);

  const handleLineClick = (time: number) => {
    onSeek(time);
  };

  return (
    <div
      className="flex flex-col h-full rounded-2xl overflow-hidden relative glass shadow-2xl bg-black/30 backdrop-blur-md"
      id="lyrics-panel-container"
    >
      {/* HEADER WITH CONTROLS */}
      <div className="flex justify-between items-center px-4 py-3 border-b border-gray-400/10 shrink-0">
        <div className="flex items-center gap-1.5">
          <span className="font-mono text-xs font-bold text-neon-pink tracking-widest uppercase">
            LYRIC SYNC PANEL
          </span>
          <span className="font-mono text-[9px] px-1.5 py-0.5 rounded bg-neon-pink/15 text-neon-pink font-semibold">
            LIVE
          </span>
        </div>

        {/* TRANSLATION SELECTION */}
        <div className="flex gap-1 bg-white/5 p-0.5 rounded-lg">
          <button
            id="btn-lyrics-both"
            onClick={() => setLyricsMode("both")}
            className={`px-2.5 py-1 rounded text-[10px] font-mono tracking-wider transition-all uppercase ${
              lyricsMode === "both"
                ? "bg-neon-pink text-white"
                : "text-gray-400 hover:text-gray-200"
            }`}
          >
            BOTH
          </button>
          <button
            id="btn-lyrics-jp"
            onClick={() => setLyricsMode("japanese")}
            className={`px-2.5 py-1 rounded text-[10px] font-mono tracking-wider transition-all uppercase ${
              lyricsMode === "japanese"
                ? "bg-neon-pink text-white"
                : "text-gray-400 hover:text-gray-200"
            }`}
          >
            JP
          </button>
          <button
            id="btn-lyrics-en"
            onClick={() => setLyricsMode("english")}
            className={`px-2.5 py-1 rounded text-[10px] font-mono tracking-wider transition-all uppercase ${
              lyricsMode === "english"
                ? "bg-neon-pink text-white"
                : "text-gray-400 hover:text-gray-200"
            }`}
          >
            EN
          </button>
        </div>
      </div>

      {/* SCROLLING BODY */}
      <div
        ref={containerRef}
        className="flex-1 overflow-y-auto px-6 py-12 scroll-smooth select-none"
        id="lyrics-scroller"
      >
        {lyrics.length > 0 ? (
          <div className="flex flex-col gap-6 max-w-xl mx-auto">
            {lyrics.map((line, index) => {
              const isActive = index === activeIndex;
              return (
                <div
                  key={index}
                  ref={isActive ? activeLineRef : null}
                  onClick={() => handleLineClick(line.time)}
                  className={`group flex flex-col gap-1 text-left cursor-pointer transition-all duration-300 origin-left ${
                    isActive
                      ? "scale-105 opacity-100 font-medium"
                      : "opacity-35 hover:opacity-75 scale-95"
                  }`}
                >
                  {/* Japanese/Romaji Lyric */}
                  {(lyricsMode === "both" || lyricsMode === "japanese") && (
                    <p
                      className={`font-sans text-lg tracking-wide md:text-xl transition-all ${
                        isActive
                          ? theme === "dark"
                            ? "text-neon-cyan drop-shadow-[0_0_10px_rgba(0,243,255,0.4)]"
                            : "text-retro-terracotta"
                          : "text-white"
                      }`}
                    >
                      {line.text}
                    </p>
                  )}

                  {/* English Lyric */}
                  {(lyricsMode === "both" || lyricsMode === "english") && line.translation && (
                    <p
                      className={`font-serif text-sm italic transition-all ${
                        isActive
                          ? theme === "dark"
                            ? "text-white"
                            : "text-retro-navy font-semibold"
                          : "text-gray-400"
                      }`}
                    >
                      {line.translation}
                    </p>
                  )}
                </div>
              );
            })}
          </div>
        ) : (
          <div className="h-full flex flex-col items-center justify-center gap-2 text-center text-gray-500 py-10">
            <Languages className="w-8 h-8 stroke-[1.2] text-gray-600 animate-pulse" />
            <p className="font-mono text-xs uppercase tracking-widest text-gray-400">
              NO LYRICS DETECTED
            </p>
            <p className="font-sans text-[11px] text-gray-500 max-w-xs">
              Load an FM City Pop cassette tape or a track from your library to view time-synced lyrics.
            </p>
          </div>
        )}
      </div>

      {/* FOOTER TIP */}
      <div className="px-4 py-2 bg-black/10 border-t border-gray-400/5 flex justify-between items-center text-[10px] font-mono text-gray-500 shrink-0">
        <div className="flex items-center gap-1">
          <HelpCircle className="w-3 h-3 text-neon-pink" />
          <span>Click any line to skip playback directly to that segment.</span>
        </div>
        <span>BPM MATCHED</span>
      </div>
    </div>
  );
}
